    /**
  	 * if you set a type for a link
  	 * we will show an icon for the link (default icon)
  	 * if you don't set imageUrl.
  	 */
	  type?: LinkMetaType
	  title: string,
	  description?: string
	  slug: string,
	  iconPicture?: string,

  		/**
